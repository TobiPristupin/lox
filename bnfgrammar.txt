GRAMMAR ADJUSTED FOR PRECEDENCE AND ASSOCIATIVITY:

program :== declaration* EOF ;

//statements
declaration :== varStatement | statement ;
varStatement :== "var" IDENTIFIER ("=" expression)? ";" ;
statement :== exprStatement | ifStatement | printStatement | block;
exprStatement :== expression ";" ;
printStatement :== "print" expression ";" ;
ifStatement :== "if" "(" expression ")" statement ("elif" "(" expression ")" statement)* ("else" statement)? ;
block = "{" declaration* "}" ;

//expressions
expression :== assignment
assignment :== (IDENTIFIER "=" equality ";") | equality;
equality :== comparison ( ("==" | "!=") comparison )* ;
comparison :== addition ( (">" | ">=" | "<" | "<=") addition )* ;
addition :== multiplication ( ("-" | "+") multiplication )* ;
multiplication :== unary ( ("/" | "*") unary )* ;
unary :== ("!" | "-") unary | primary ;
primary :== NUMBER | STRING | IDENTIFIER | "false" | "true" | "nil" | "(" expression ")" ;



SIMPLIFIED GRAMMAR:

program :== declaration* EOF ;

//statements
declaration :== varStatement | statement ;
varStatement :== "var" IDENTIFIER ("=" expression)? ";" ;
statement :== exprStatement | printStatement | ifStatement | block ;
exprStatement :== expression ";" ;
printStatement :== "print" expression ";" ;
ifStatement :== "if" "(" expression ")" statement ("elif" "(" expression ")" statement)* ("else" statement)? ;
block = "{" declaration* "}" ;

//expressions
assignment :== IDENTIFIER "=" expression ";" ;
expression :== literal | unary | binary | grouping ;
literal :== NUMBER | STRING | "true" | "false" | "nil" ;
unary :== ("-" | "!") expression ;
grouping :== "(" expression ")" ;
binary :== expression operator expression
operator:== "==" | "!=" | "<" | "<=" | ">" | ">=" | "+"  | "-"  | "*" | "/" ;
